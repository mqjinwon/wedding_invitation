// Firestore 보안 규칙
// Firebase Console > Firestore Database > 규칙 탭에 복사하여 붙여넣으세요

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // 방명록 컬렉션에 대한 규칙
    match /guestbooks/{document} {
      // 모든 사용자가 읽기 가능
      allow read: if true;
      
      // 모든 사용자가 쓰기 가능 (방명록 작성)
      allow create: if true;
      
      // 수정과 삭제는 해당 문서의 비밀번호를 알고 있어야 함
      // (클라이언트에서 검증하므로 일단 허용)
      allow update, delete: if true;
    }
    
    // 다른 모든 문서는 기본적으로 거부
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
